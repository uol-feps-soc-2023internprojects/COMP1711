Extension

Exercise 2 - Patterns!

Take this program, pattern.c from the notes.

#include <stdio.h>
int main()
{
    for(int i = 1; i < 8; i++)
    {
        for(int j = 0; j < i; j++)
        {
            printf("*");
        }
        printf("\n");
    }
}

With this code the program will output:

*
**
***
****
*****
******
*******

Alter the program so that it outputs these different patterns:

1

*****
****
***
**
*

Answer

#include <stdio.h>
int main()
{
    for(int i = 6; i > 1; i--)
    {
        for(int j = 1; j < i; j++)
        {
            printf("*");
        }
        printf("\n");
    }
}

2

   *
  ***
 *****

Answer

#include <stdio.h>
int main()
{
    for(int i = 1; i < 6; i+=2)
    {
        for(int j = 0; j < 7-i; j+=2)
        {
            printf(" ");
        }
        for(int j = 0; j < i; j++)
        {
            printf("*");
        }
        printf("\n");
    }
}

3

Make this set of patterns that could create an animation:

*****
*   *
*   *
*   *
*****
-----
 * * 
*****
 * * 
*****
 * * 
-----
  *  
  *  
*****
  *  
  *  
-----
 * * 
*****
 * * 
*****
 * * 
-----
*****
*   *
*   *
*   *
*****
-----

Answer

#include <stdio.h>
#define GRID_SIZE 5
#define MAX_INDEX GRID_SIZE-1

int main()
{
    for(int k = 0; k < GRID_SIZE; k++)
    {
        for(int i = 0; i < GRID_SIZE; i++)
        {
            for(int j = 0; j < GRID_SIZE; j++)
            {
                if(i == 0 + k || i == MAX_INDEX-k || j == 0 + k | j  == MAX_INDEX - k)
                {
                    printf("*");
                }
                else
                {
                    printf(" ");
                }
            }
            printf("\n");
        }
        printf("-----\n");
    }
    return 0;
}
